on_actions = {
	on_startup = {
		effect = {
			every_country = {
				set_variable = { GER_Influence = 0 }
				clamp_variable = {
					var = GER_Influence
					min = 0
					max = 100
				}					
			}
		}
	}
	on_weekly_HUN = {
		effect = {
			every_country = {
				add_to_variable = { GER_Influence = modifier@weekly_influence_growth }
				clamp_variable = {
					var = GER_Influence
					min = 0
					max = 100
				}	
			}
		}
	}
	on_weekly_YUG = {
		effect = {
			every_country = {
				add_to_variable = { GER_Influence = modifier@weekly_influence_growth }
				clamp_variable = {
					var = GER_Influence
					min = 0
					max = 100
				}	
			}
		}
	}
	on_weekly_BUL = {
		effect = {
			every_country = {
				add_to_variable = { GER_Influence = modifier@weekly_influence_growth }
				clamp_variable = {
					var = GER_Influence
					min = 0
					max = 100
				}	
			}
		}
	}
	on_weekly_ROM = {
		effect = {
			every_country = {
				add_to_variable = { GER_Influence = modifier@weekly_influence_growth }
				clamp_variable = {
					var = GER_Influence
					min = 0
					max = 100
				}	
			}
		}
	}
}
