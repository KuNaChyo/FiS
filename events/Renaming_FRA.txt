# FRA - rename 3 - "Suisse"
country_event = {
	id = fra_rename.3
	hidden = yes

	trigger = {
		tag = FRA
		controls_state = 3
		NOT = { has_country_flag = renamed_state_3 }
	}

	mean_time_to_happen = {
		days = 3
	}

	immediate = {
		hidden_effect = {
			every_country = {
				limit = { has_country_flag = renamed_state_3 }
				clr_country_flag = renamed_state_3
			}
		}
	}

	option = {
		name = fra_rename.3.a
		3 = { set_state_name = "Suisse" } 
		set_province_name = { id = 9620 name = "Berne"}
		set_country_flag = renamed_state_3
		set_global_flag = renamed_state_3
	}
}
