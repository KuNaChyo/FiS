### Reset - rename 1
country_event = {
	id = europe_reset_rename.1
	hidden = yes

	trigger = {
		NOT = { OR = { tag = AUS tag = GER  is_subject_of = GER } }
		controls_state = 1
		NOT = { has_country_flag = renamed_state_1 }
		# This state has been renamed by another country
		has_global_flag = renamed_state_1
	}

	mean_time_to_happen = {
		days = 3
	}

	immediate = {
		hidden_effect = {
			every_country = {
				limit = { has_country_flag = renamed_state_1 }
				clr_country_flag = renamed_state_1
			}
		}
	}

	option = {
		name = europe_reset_rename.1.a
		1 = { reset_state_name = yes }
		reset_province_name = 3838 

		set_country_flag = renamed_state_1
		clr_global_flag = renamed_state_1
	}
}

	id = europe_reset_rename.176
	hidden = yes

	trigger = {
		NOT = { OR = { tag = AUS tag = GER  is_subject_of = GER } }
		controls_state = 176
		NOT = { has_country_flag = renamed_state_176 }
		# This state has been renamed by another country
		has_global_flag = renamed_state_176
	}

	mean_time_to_happen = {
		days = 3
	}

	immediate = {
		hidden_effect = {
			every_country = {
				limit = { has_country_flag = renamed_state_176 }
				clr_country_flag = renamed_state_176
			}
		}
	}

	option = {
		name = europe_reset_rename.176.a
		176 = { reset_state_name = yes }
		reset_province_name = 885
		reset_province_name = 11825
		set_country_flag = renamed_state_176
		clr_global_flag = renamed_state_176
	}
}
